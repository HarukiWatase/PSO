        \begin{tcolorbox}[colback=white, colframe=white, boxrule=0pt, left=0pt, right=0pt, top=5pt, bottom=5pt]
            \normalsize
            \textbf{■ 社会的背景}
            \begin{itemize}
                \item 5G、IoTの普及によるネットワークトラフィックの増大と多様化
                \item 自動運転や遠隔医療など、高品質な通信を要するサービスの台頭
                \item 従来の最短経路のみでは対応困難な、多様なQoS要求の発生
            \end{itemize}
            \vspace{0.5em}
            \textbf{■ 多制約最適経路問題}
            \begin{itemize}
                \item ボ\hspace{0.01em}ト\hspace{-0.2em}ルネック帯域最大化および遅延等制約を満たす経路探索
                \item 本問題は\textcolor{AlertRed}{\textbf{NP困難}}
                \item {厳密解法 [1]}: 最適性保証の一方、規模・制約数増加による\textbf{計算時間の指数関数的爆発}
            \end{itemize}
            \vspace{0.5em}
            \textbf{■ 従来手法の課題}
            \begin{itemize}
                \item {従来のPSO (Particle Swarm Optimization) [2]}: 全粒子での情報共有による、複雑制約下での\textbf{局所解への早期収束}
            \end{itemize}
            
            \vspace{0.5em}
            
            \textbf{■ 本研究の目的}
            \begin{itemize}
                \item \textbf{高速化とスケーラビリティ}: 厳密解法が破綻する大規模・多制約環境でも、実用的な計算時間で解を算出
                \item \textbf{ロバスト\hspace{-0.2em}性の向上}: 従来PSOが陥る局所解停滞を回避し、厳しい制約条件下でも粘り強く解を発見
                \item \textbf{提案手法}: 上記を両立する\textcolor{AlertRed}{\textbf{「並列化改良型PSO」}}を提案し、その有効性を実証
            \end{itemize}
        \end{tcolorbox}

        \vspace{0.3cm}

        \begin{block}{2. 基礎手法: 粒子群最適化 (PSO)}
            \begin{columns}[T, totalwidth=\linewidth]
                \begin{column}{0.52\linewidth}
                    \normalsize
                    \textbf{■ 概要}
                    \begin{itemize}
                        \item 生物の群れ行動を模倣した群知能アルゴリズム
                        \item 粒子が探索空間を移動し解を探索
                    \end{itemize}
                    \vspace{0.5em}
                    \textbf{■ 更新メカニズム}
                    \begin{itemize}
                        \item Personal Best: 自身の過去最良
                        \item Global Best: 群れ全体の最良
                        \item これらを頼りに速度$\boldsymbol{v}$と位置$\boldsymbol{x}$を更新
                    \end{itemize}
                \end{column}
                \begin{column}{0.55\linewidth}

                        \item 制約数3（遅延・ロス・信頼性）/ 制約数1（遅延）
                    \end{itemize}
                    \vspace{0.3em}
                    \textbf{■ 結果と考察}
                    \begin{itemize}
                        \item {厳密解法[1]}: 制約数増加に伴う、\textbf{計算時間の指数関数的増大}
                        \item {提案手法}: 3制約下でノード数増加につれて、\textbf{厳密解法より高速}
                        \item 並列化: 4並列化により、2.91倍高速化成功（理論値: 3.07倍）
                    \end{itemize}
